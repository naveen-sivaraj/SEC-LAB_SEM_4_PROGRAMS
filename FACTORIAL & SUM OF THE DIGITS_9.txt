Factorial

declare
i number(4):=1;
n number(4):=&n;
f number(4):=1;
begin
for i in 1..n loop
f:=f*i;
end loop;
Dbms_output.put_line('The Factorial of '||n||' is:'||f);
end;
/

Sum for digits

Declare
n number:=&n;
s number:=0;
r number:=0;
begin
while n !=0
loop
r:=mod(n,10);
s:=s+r;
n:=trunc(n/10);
end loop;
dbms_output.put_line('Sum of digits of given number is '||s);
end;
